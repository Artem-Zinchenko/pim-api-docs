Get a single product

REQUEST

GET api/rest/v1/products/my_product HTTP/1.1
Host: demo.akeneo.com
Content-Type: application/json; charset=utf-8

RESPONSE

200 OK

{
  "identifier": "myproduct",
  "family": null,
  "groups": ["group_code_1", "group_code_3"],
  "variant_group": null,
  "categories": ["category_code_1"],
  "values": {
    "attribute_code_1":[
      {
        "locale": null,
        "scope": null
        "data": 10
      }
    ]
  },
  "associations": {
    "pack": {
      "groups": []
      "products": ["product_identifier_1","product_identifier_2"]
     },
     "x_sell": {
       "groups": [],
       "products": []
     }
  }
}

Update a family

REQUEST

PATCH api/rest/v1/families/myfamily HTTP/1.1
Host: demo.akeneo.com
Content-Type: application/json; charset=utf-8

{
  "code":"myfamily",
  "labels":{
       "en_US":"Family 1",
       "de_DE":"Familie 1",
       "fr_FR":"Famille 1"
   },
  "attributes":["attribute1", "attribute2", "attribute3"],
  "attribute_as_label":"attribute1",
  "attribute_requirements":{
    "channel1":["attribute1", "attribute3"]
  }
}

RESPONSE

204 No Content
